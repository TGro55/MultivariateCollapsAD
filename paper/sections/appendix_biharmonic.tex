Writing out the multivariate chain-rule of the Biharmonic Operator applied to $g \circ h$ we see several terms which occur only as sums (marked in \mathemph{blue}):
\begin{align}
 \label{eq:biharm-multi}
&\sum_{i=1}^D \sum_{j=1}^D 
\partial^4 g(h) [h^i, h^i, h^j, h^j] \\
&+ 
\sum_{i=1}^D  \partial^3 g(h) \left[
 h^i, h^i, 
\mathemph{\sum_{j=1}^D  h^{jj}}
\right]
+
4 \sum_{i=1}^D \sum_{j=1}^D \partial^3 g(h) \left[
h^i, h^j, h^{ij}
\right] \\
&+
\sum_{j=1}^D  \partial^3 g(h) \left[
\mathemph{\sum_{i=1}^D  h^{ii}}, h^j, h^j
\right] 
+
2 \sum_{i=1}^D \sum_{j=1}^D \partial^2 g(h) \left[
h^{ij}, h^{ij}
\right] \\
&+
2 \sum_{i=1}^D  \partial^2 g(h) \left[
h^i, \mathemph{\sum_{j=1}^D  h^{ijj}}
\right] 
+
2 \sum_{j=1}^D \partial^2 g(h) \left[
\mathemph{\sum_{i=1}^D  h^{iij}}, h^j 
\right] \\
&+ 
\partial^2 g(h) \left[
\mathemph{\sum_{i=1}^D h^{ii}, \sum_{j=1}^D  h^{jj}}
\right] 
 +
\partial g(h) \left[
\mathemph{\sum_{i=1}^D \sum_{j=1}^D h^{iijj}}
\right]
\end{align}
There are several terms that are equivalent, like $\displaystyle \sum_j h^{jj} = \sum_i h^{ii}$, $\displaystyle \sum_j h^{ijj} = \sum_i h^{iij}$ or $\displaystyle \sum_{i=1}^d  \sum_{j=1}^d \partial^3 g(h) [h^i, h^i, h^{jj}] = \displaystyle \sum_{i=1}^d  \sum_{j=1}^d \partial^3 g(h) [h^{ii}, h^j, h^j]$. All simplifications lead to the following multivariate propagation scheme
\begin{align}
\label{eq:biharm-efficient}
  \begin{split}
    \begin{pmatrix*}
      \vx^0
      \\
      \{\vx^{i} \}
      \\
      \{\vx^{ij} \}
      \\
      \mathemph{\displaystyle\sum_{i=1}^D \vx^{ii}}  
      \\
      \left\{ \mathemph{\displaystyle\sum_{i=1}^D \vx^{iij}} \right\}
      \\
      \mathemph{\displaystyle\sum_{i=1}^D \sum_{j=1}^D  \vx^{iijj}}
    \end{pmatrix*}
    &\overset{\text{(\ref{eq:taylor-mode-scalar})}}{\to}
      \begin{pmatrix*}[l]
        \vh^0 =  \vh(\vx^0)
        \\
        \{\vh^{i}\} = \{\partial \vh \left[\vx^{i} \right]\}
        \\
        \{\vh^{ij}\} = \{\partial^2 \vh \left[\vx^{i}, \vx^j \right] 
        + \partial \vh \left[ \vx^{ij} \right]\}
        \\
        \mathemph{\displaystyle \sum_{i=1}^D \vh^{ii}} = \displaystyle\sum_{i=1}^D \partial^2 \vh \left[
        \vx^{i}, \vx^{i}
        \right]
        + \partial \vh \left[
        \mathemph{\displaystyle\sum_{i=1}^D\vx^{ii}}
        \right]
        \\
        \left\{\mathemph{\displaystyle \sum_{i=1}^D \vh^{iij}}
        \right\} 
        = 
        \left\{\displaystyle\sum_{i=1}^D 
        \partial^3 \vh \left[
        \vx^{i}, \vx^{i}, \vx^{j}
        \right] 
        + 
        2 \displaystyle\sum_{i=1}^D \partial^2 \vh \left[
        \vx^{ij}, \vx^{i} 
        \right]
        + 
        \partial^2 \vh \left[
        \vx^{j}, \mathemph{\displaystyle\sum_{i = 1}^D\vx^{ii}}
        \right]
        \right. 
        \\
        \left. 
        + 
        \partial \vh \left[
        \mathemph{\displaystyle\sum_{i=1}^D\vx^{iij}}
        \right] 
        \right\}
        \\
        \mathemph{\displaystyle\sum_{i=1}^D \sum_{j=1}^D \vh^{iijj}} = 
        \displaystyle\sum_{i=1}^D \sum_{j=1}^D
        \partial^4 \vh \left[
        \vx^i, \vx^i, \vx^j, \vx^j 
        \right] 
        + 
        2 \displaystyle \sum_{j=1}^D \partial^3 \vh \left[
        \vx^j, \vx^j, \mathemph{\displaystyle \sum_{i=1}^D \vx^{ii}} 
        \right]
        \\
        +
        4 \displaystyle \sum_{i=1}^D \sum_{j=1}^D \partial^3 \vh \left[
        \vx^i, \vx^j, \vx^{ij}
        \right]
        +
        \displaystyle 2\sum_{i=1}^D \sum_{j=1}^D \partial^2 \vh \left[
         \vx^{ij}, \vx^{ij}
         \right]
         \\
        +
        \displaystyle 4\sum_{j=1}^D \partial^2 \vh \left[
        \vx^j, \mathemph{\sum_{i=1}^D  \vx^{iij}}
        \right]
        +
        \displaystyle \partial^2 \vh \left[
        \mathemph{\sum_{i=1}^D \vx^{ii}}, \mathemph{\sum_{i=1}^D \vx^{ii}}
        \right]
        + 
        \partial \vh \left[
        \mathemph{\sum_{i=1}^D \sum_{j=1}^D \vx^{iijj}}
        \right]
      \end{pmatrix*}
        \\
    &\overset{\text{(\ref{eq:taylor-mode-scalar})}}{\to}
        \begin{pmatrix*}[l]
        \vg^0 =  \vg(\vh^0)
        \\
        \{\vg^{i}\} = \{ \partial \vg \left[ \vh^{i} \right]\}
        \\
       \{\vg^{ij}\} = \{\partial^2 \vg \left[\vh^{i}, \vh^j \right] 
        + \partial \vg \left[ \vh^{ij} \right]\}
        \\
        \mathemph{\displaystyle \sum_{i=1}^D \vg^{ii}} = \displaystyle\sum_{i=1}^D \partial^2 \vg \left[
        \vh^{i}, \vh^{i}
        \right]
        + \partial \vg \left[
        \mathemph{\displaystyle\sum_{i=1}^D\vh^{ii}}
        \right]
        \\
        \left\{ \mathemph{\displaystyle \sum_{i=1}^D \vg^{iij}}
        \right\}
        = 
        \left\{ \displaystyle\sum_{i=1}^D
        \partial^3 \vg \left[
        \vh^{i}, \vh^{i}, \vh^{j}
        \right] 
        + 
        2 \displaystyle\sum_{i=1}^D
        \partial^2 \vg \left[
        \vh^{ij}, \vh^{i}
        \right]
         + 
        \partial^2 \vg \left[
        \vh^{j}, \mathemph{\displaystyle\sum_{i=1}^D\vh^{ii}}
        \right] 
        \right.
        \\
        \left.
        + 
        \partial \vg \left[
        \mathemph{\displaystyle\sum_{i=1}^D\vh^{iij}} 
        \right]
        \right\}
        \\
        \mathemph{\displaystyle\sum_{i=1}^D \sum_{j=1}^D \vg^{iijj}} = 
        \displaystyle\sum_{i=1}^D \sum_{j=1}^D
        \partial^4 \vg \left[\vh^i, \vh^i, \vh^j, \vh^j \right] 
        + 
        2 \displaystyle \sum_{j=1}^D \partial^3 \vg \left[
        \vh^j, \vh^j, \mathemph{\displaystyle \sum_{i=1}^D \vh^{ii}} 
        \right]
        \\
        +
        4 \displaystyle \sum_{i=1}^D \sum_{j=1}^D \partial^3 \vg \left[
        \vh^i, \vh^j, \vh^{ij}
        \right]
        +
        \displaystyle 2\sum_{i=1}^D \sum_{j=1}^D \partial^2 \vg \left[
         \vh^{ij}, \vh^{ij}
         \right] \\
        +
        4 \displaystyle \sum_{j=1}^D \partial^2 \vg \left[
        \vh^j, \mathemph{\sum_{i=1}^D  \vh^{iij}}
        \right]
        +
        \displaystyle \partial^2 \vg \left[
        \mathemph{\sum_{i=1}^D \vh^{ii}}, \mathemph{\sum_{i=1}^D \vh^{ii}}
        \right]
        + 
        \partial \vg \left[
        \mathemph{\sum_{i=1}^D \sum_{j=1}^D \vh^{iijj}}
        \right]
      \end{pmatrix*} 
    \\
    &\overset{\text{slice}}{\to} 
    \mathemph{\sum_{i=1}^D \sum_{i=1}^D \{ \vg^{iijj} \}}
      \overset{ \{ \partial \vx = I_D, \partial^2 \vx = 0, \partial^3 \vx = 0, \partial^4 \vx = 0 \} }{=}
      \Delta^2 \vf(\vx_0)
  \end{split}
\end{align}
